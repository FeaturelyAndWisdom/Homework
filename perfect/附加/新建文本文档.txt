module A(clk2,key,rst,enter,Q,count);
input clk2,key,rst,enter;
output reg[31:0] Q;
output reg [3:0] count;

always@(posedge clk2 or negedge rst)
	begin 
		if(!rst)
		   count<=4'd0;
		else  
			if( !key || !rst || !enter)
				begin
				count<=0;
				Q<=32'h00000000; end
			else
			  begin
			     count<=count+4'd1;
			     if(count>4'd9)
				    begin
				        Q<=32'h000000CE;
				        if (count==4'd15) count<=4'd10;
				    end
			  end
	end
	
endmodule
